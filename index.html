let dx = []; let conduta = []; let exames = []; let alertas = []; let exameExtra = ""; // RED FLAGS if(t.includes("dispneia") && t.includes("febre")){ alertas.push("🚨 Risco respiratório grave"); } if(t.includes("dor torácica")){ alertas.push("🚨 Avaliar SCA"); } if(t.includes("rebaixamento") || t.includes("confusão")){ alertas.push("🚨 Alteração neurológica"); } // DIAGNÓSTICO + CONDUTA if(t.includes("febre") && t.includes("tosse")){ dx.push("Infecção respiratória / Pneumonia"); conduta.push("Considerar ATB"); exames.push("RX tórax", "Hemograma", "PCR"); exameExtra += "AR: MVBD, avaliar crepitações.\n"; } if(t.includes("dispneia")){ dx.push("Síndrome respiratória"); conduta.push("Monitorar SatO2"); exames.push("Gasometria"); exameExtra += "AR: esforço respiratório presente.\n"; } if(t.includes("dor abdominal")){ dx.push("Abdome agudo"); conduta.push("Avaliar irritação peritoneal"); exames.push("USG abdome"); exameExtra += "ABD: avaliar dor localizada e defesa.\n"; } if(t.includes("náusea") || t.includes("vômito")){ dx.push("Síndrome gastrointestinal"); conduta.push("Hidratação + antiemético"); exames.push("Eletrólitos"); } return { dx: dx.join("; ") || "A esclarecer", conduta: conduta.join("\n- ") || "Conduta geral", exames: exames.join(", "), alertas: alertas.join("\n"), exameExtra }; } function gerarSOAP(){ let sexo = document.getElementById('sexo').value; let qp = document.getElementById('qp').value; let sintomas = document.getElementById('sintomas').value; let dados = analisar(sintomas); let exameBase = sexo === 'fem' ? `BEG, afebril, hidratada.\nNeuro: Glasgow 15.\nACV: RCR.` : `BEG, afebril, hidratado.\nNeuro: Glasgow 15.\nACV: RCR.`; let soap = `S: ${qp}\nSintomas: ${sintomas}\n\nO:\n${exameBase}\n${dados.exameExtra}\n\nA:\n${dados.dx}\n\nP:\n- ${dados.conduta}\n- Exames: ${dados.exames}\n- Retorno se piora`; if(dados.alertas){ soap = `🚨 ALERTAS:\n${dados.alertas}\n\n` + soap; } document.getElementById('resultado').value = soap; } function copiar(){ navigator.clipboard.writeText(document.getElementById('resultado').value); } function limpar(){ document.getElementById('qp').value = ''; document.getElementById('sintomas').value = ''; document.getElementById('resultado').value = ''; }
